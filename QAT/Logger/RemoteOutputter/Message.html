def initialize(log_event, level, level_name, formatter)
  @log_event                 = log_event
  @content                   = {}
  @content[:level]           = level
  @content['_level_name']    = level_name
  @content["_logger"]        = log_event.fullname
  @content["_log_timestamp"] = Time.now.iso8601(3)

  @formatter = formatter

  set_full_message
  set_short_message
end
